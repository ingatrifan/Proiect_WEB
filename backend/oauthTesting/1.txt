1)CREATE LOBBY
CALL
###
	PROTOCOL : CREATE_LOBBY
	PLAYER_NAME : some_name
	ROOM_NAME : some_name
	MAX_PLAYERS : some_nr
	
###RESPONSE 
	PROTOCOL : CREATE_LOBBY
	ANSWER : BAD CALLING/ Room Created Successfully
	SUCCESS : 0/1 
	userList : [] - lista cu Useri din lobby ()
	name: - numele care l-ati dat mai sus
	dungeonMaster : - cine ii dungeon master( e cel ce apeleaza metoda createLobby la inceput)
	id : object id-ul loby-ului
	maxPlayersLimit : limita setata
	
	eg raspuns : {"userList":[],"SUCCESS":"1","ANSWER":"Room Created Successfully","name":"broh","PROTOCOL":"CREATE_LOBBY","id":"5ea8867c2a50cf30546b9d64","maxPlayersLimit":100,"dungeonMaster":"someName"}
